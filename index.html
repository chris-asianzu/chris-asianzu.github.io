SQL Mastery

Business-First SQL

Every query answers a real business question. Here are three examples from the repository.

Pareto Revenue Analysis Revenue
-- Which products drive 80% of revenue? WITH cumulative AS ( SELECT product_id, SUM(revenue) AS total, SUM(SUM(revenue)) OVER (ORDER BY SUM(revenue) DESC) AS cumulative_rev FROM orders GROUP BY product_id ) SELECT product_id, ROUND(cumulative_rev / grand_total * 100, 2) AS cumulative_pct FROM cumulative WHERE cumulative_rev / grand_total <= 0.80;
RFM Customer Scoring Customers
-- Score every customer on R, F, M WITH rfm_scored AS ( SELECT customer_id, NTILE(5) OVER (ORDER BY recency_days ASC) AS r_score, NTILE(5) OVER (ORDER BY frequency DESC) AS f_score, NTILE(5) OVER (ORDER BY monetary DESC) AS m_score FROM rfm_base ) SELECT *, r_score + f_score + m_score AS total_rfm_score FROM rfm_scored ORDER BY total_rfm_score DESC;
MoM Revenue Growth Revenue
-- Month-over-month growth with LAG SELECT DATE_TRUNC('month', order_date) AS order_month, SUM(order_value) AS revenue, LAG(SUM(order_value)) OVER (ORDER BY DATE_TRUNC('month', order_date)) AS prev_month, ROUND((revenue - prev_month) / prev_month * 100, 2) AS mom_growth_pct FROM orders GROUP BY order_month ORDER BY order_month;
