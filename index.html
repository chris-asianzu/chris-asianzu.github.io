RFM Customer ScoringCustomers
-- Who are your best customers? WITH rfm_scored AS ( SELECT customer_id, NTILE(5) OVER ( ORDER BY recency_days ASC) AS r, NTILE(5) OVER ( ORDER BY frequency DESC) AS f, NTILE(5) OVER ( ORDER BY monetary DESC) AS m FROM rfm_base ) SELECT *, r+f+m AS rfm_score, CASE WHEN r>=4 AND f>=4 THEN 'Champion' ELSE 'Needs Review' END AS segment FROM rfm_scored ORDER BY rfm_score DESC;
MoM Revenue GrowthRevenue
-- Is revenue growing month over month? SELECT DATE_TRUNC('month', order_date) AS month, SUM(order_value) AS revenue, LAG(SUM(order_value)) OVER ( ORDER BY DATE_TRUNC( 'month', order_date)) AS prev_revenue, ROUND( (SUM(order_value) - LAG(SUM(order_value)) OVER (ORDER BY DATE_TRUNC( 'month',order_date))) / NULLIF(LAG(SUM(order_value)) OVER (ORDER BY DATE_TRUNC( 'month',order_date)),0)*100 ,2) AS mom_pct FROM orders GROUP BY month;
Cohort Retention RateCohort
-- What % of each cohort stays? WITH first_purchase AS ( SELECT customer_id, DATE_TRUNC('month', MIN(order_date)) AS cohort_month FROM orders GROUP BY customer_id ), cohort_size AS ( SELECT cohort_month, COUNT(*) AS total FROM first_purchase GROUP BY cohort_month ) SELECT cohort_month, month_number, ROUND(retained * 100.0 / total, 2) AS retention_pct FROM activity a JOIN cohort_size cs ON a.cohort_month = cs.cohort_month ORDER BY cohort_month, month_number;
